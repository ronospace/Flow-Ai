   642	  Future<void> _openEmailInvite() async {
   643	    if (_isExecutingInvite || _isSent || _isLoading) return;
   644	    final email = _emailController.text.trim();
   645	    final message = _messageController.text.trim();
   646	
   647	    if (_successMessage != null) {
   648	      setState(() => _successMessage = null);
   649	    }
   650	
   651	    if (!_formKey.currentState!.validate()) {
   652	      return;
   653	    }
   654	
   655	    setState(() {
   656	      _isExecutingInvite = true;
   657	      _isLoading = true;
   658	    });
   659	
   660	    try {
   661	      final result = await widget.partnerService.sendPartnerInvitation(
   662	        inviteeEmail: email,
   663	        personalMessage: message.isEmpty ? null : message,
   664	      );
   665	
   666	      if (!mounted) return;
   667	
   668	      if (result == null) {
   669	        _showErrorMessage("Failed to send invitation");
   670	        return;
   671	      }
   672	
   673	      setState(() {
   674	        _successMessage = "Invitation sent successfully ❤️";
   675	        _isSent = true;
   676	      });
   677	
   678	      Future.delayed(const Duration(seconds: 2), () {
   679	        if (mounted) {
   680	          setState(() => _successMessage = null);
   681	          Navigator.of(context).pop();
   682	        }
   683	      });
   684	    } catch (e) {
   685	      _showErrorMessage("Error sending invitation: $e");
   686	    } finally {
   687	      if (mounted) {
   688	        setState(() {
   689	          _isLoading = false;
   690	          _isExecutingInvite = false;
   691	        });
   692	      } else {
   693	        _isExecutingInvite = false;
   694	      }
   695	    }
   696	  }
