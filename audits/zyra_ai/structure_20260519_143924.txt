   192	  void _toggleChat() {
   193	    setState(() {
   194	      _isExpanded = !_isExpanded;
   195	      if (!_isExpanded) {
   196	        _isFullScreen = false;
   197	        _chatHeight = 0.5;
   198	      }
   199	    });
   200	
   201	    if (_isExpanded) {
   202	      _stopPulsingAnimation();
   203	      _chatController.forward();
   204	    } else {
   205	      _chatController.reverse();
   206	      _expandController.reverse();
   207	      _startPulsingAnimation();
   208	    }
   209	  }
   210

   213	      _isFullScreen = !_isFullScreen;

   215	    });
   216	
   217	    if (_isFullScreen) {
   218	      _expandController.forward();
   219	    } else {
   220	      _expandController.reverse();
   221	    }
   222	  }

   266	  double _getTabsBottom() {
   267	    final ctx = widget.tabsKey.currentContext;
   268	    if (ctx == null) return 200;
   269	    final box = ctx.findRenderObject() as RenderBox;
   270	    final pos = box.localToGlobal(Offset.zero);
   271	    return pos.dy + box.size.height + 8;
   272	  }
   273	
   274	  double _getPeriodSelectorTop() {
   275	    final ctx = widget.periodSelectorKey.currentContext;
   276	    if (ctx == null) return MediaQuery.of(context).padding.top + 80;
   277	    final box = ctx.findRenderObject() as RenderBox;
   278	    final pos = box.localToGlobal(Offset.zero);
   279	    return pos.dy;
   280	  }

   983	  Widget _buildLiveStatusIndicator() {
   984	    final isThinking = _isTyping;
   985	    final coreColor = isThinking ? AppTheme.primaryPurple : AppTheme.accentMint;
   986	    final haloColor = isThinking
   987	        ? AppTheme.primaryRose
   988	        : AppTheme.secondaryBlue;
   989	    final pulseDuration = isThinking ? 1050.ms : 1600.ms;
   990	    final haloBegin = isThinking ? 0.82 : 0.76;
   991	    final haloEnd = isThinking ? 1.28 : 1.18;
   992	    final dotBegin = isThinking ? 0.94 : 0.97;
   993	    final dotEnd = isThinking ? 1.09 : 1.05;
   994	
   995	    return Stack(
   996	      alignment: Alignment.center,
   997	      children: [
   998	        Container(
   999	              width: 16,
  1000	              height: 16,
  1001	              decoration: BoxDecoration(
  1002	                shape: BoxShape.circle,
  1003	                color: haloColor.withValues(alpha: isThinking ? 0.20 : 0.18),
  1004	              ),
  1005	            )
  1006	            .animate(onPlay: null)
  1007	            .scaleXY(
  1008	              begin: haloBegin,
  1009	              end: haloEnd,
  1010	              duration: pulseDuration,
  1011	              curve: Curves.easeInOutCubic,
  1012	            )
  1013	            .fadeIn(duration: 180.ms)
  1014	            .then(delay: 120.ms)
  1015	            .fadeOut(duration: isThinking ? 520.ms : 900.ms),
  1016	
  1017	        Container(
  1018	              width: 9,
  1019	              height: 9,
  1020	              decoration: BoxDecoration(
  1021	                shape: BoxShape.circle,
  1022	                gradient: isThinking
  1023	                    ? const LinearGradient(
  1024	                        colors: [AppTheme.primaryRose, AppTheme.primaryPurple],
  1025	                        begin: Alignment.topLeft,
  1026	                        end: Alignment.bottomRight,
  1027	                      )
  1028	                    : const LinearGradient(
  1029	                        colors: [AppTheme.secondaryBlue, AppTheme.accentMint],
  1030	                        begin: Alignment.topLeft,
  1031	                        end: Alignment.bottomRight,
  1032	                      ),
  1033	                color: null,
  1034	                boxShadow: [
  1035	                  BoxShadow(
  1036	                    color: coreColor.withValues(
  1037	                      alpha: isThinking ? 0.46 : 0.44,
  1038	                    ),
  1039	                    blurRadius: isThinking ? 13 : 10,
  1040	                    spreadRadius: isThinking ? 1.8 : 1.4,
  1041	                  ),
  1042	                ],
  1043	              ),
  1044	            )
  1045	            .animate(onPlay: null)
  1046	            .scaleXY(
  1047	              begin: dotBegin,
  1048	              end: dotEnd,
  1049	              duration: pulseDuration,
  1050	              curve: Curves.easeInOutCubic,
  1051	            ),
  1052	      ],
  1053	    );
